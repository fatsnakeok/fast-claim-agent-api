支付签单 + 人工审批 US-U5 · approveQuote 
flowchart TD
    subgraph Pay [支付签单流程]
      P1([POST /api/insurance/pay
{quoteId}]) --> P_Auth{权限校验}
      P_Auth -- 无权限 --> P_403([403])
      P_Auth -- 通过 --> P_Find[quoteRepository.findById]
      P_Find --> P_Exist{报价单存在?}
      P_Exist -- 否 --> P_400a([400 报价单不存在])
      P_Exist -- 是 --> P_Status{status == APPROVED?}
      P_Status -- 否 --> P_400b([400 仅已批准的报价单可支付])
      P_Status -- 是 --> P_Expire{未过期?}
      P_Expire -- 否 --> P_400c([400 报价单已过期])
      P_Expire -- 是 --> P_Pay[模拟支付处理
log.info]
      P_Pay --> P_Create[创建 Policy
policyNumber 自动生成
effectiveDate=now
expirationDate=+lengthOfYear
status=ACTIVE]
      P_Create --> P_Save[policyRepository.save]
      P_Save --> P_OK([200 PolicyResponse])
    end

    subgraph Approve [人工审批流程]
      A1([POST /quotes/{id}/approve

      A_Auth -- 无权限 --> A_403([403])
      A_Auth -- 通过 --> A_Find[quoteRepository.findById]
      A_Find --> A_Exist{存在?}
      A_Exist -- 否 --> A_400a([400 报价单不存在])
      A_Exist -- 是 --> A_Status{status == REFERRED?}
      A_Status -- 否 --> A_400b([400 仅REFERRED可审批])
      A_Status -- 是 --> A_Expire{未过期?}
      A_Expire -- 否 --> A_400c([400 报价单已过期])
      A_Expire -- 是 --> A_Override{overridePremium

      A_Override -- 是 --> A_Set[quote.premiumAmount = override]
      A_Override -- 否 --> A_Approve
      A_Set --> A_Approve[quote.status = APPROVED
quoteRepository.save]
      A_Approve --> A_OK([200 ApproveQuoteResponse])
    end
